# Calculadora_de_media_escolar
const nota1 = 7;

const nota2 = 5;

const nota3 = 6.5;

const media = (nota1 + nota2 + nota3) / 3;

if (media < 5) {

console.log ('Está de Reprovado')

} else if (media >= 5 && media < 7){

console.log ('Está de Recuperação')

}else {

console.log ('Está aprovado')

}
